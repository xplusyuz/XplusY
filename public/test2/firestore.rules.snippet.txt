// Firestore Rules snippet for test_code results (optional)
// This allows ONLY signed-in users to write their own result documents.
// If you use your own token-based API (recommended), you don't need this.
match /test_code/{testCode}/results/{uid} {
  allow read: if true; // or restrict as needed
  allow write: if request.auth != null && request.auth.uid == uid;
}
match /test_code/{testCode}/attempts/{attemptId} {
  allow read: if true; // or restrict
  // if you allow client writes:
  allow create: if request.auth != null;
  allow update, delete: if false;
}
